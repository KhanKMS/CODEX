4. Docker Machine 설치 필요 
   $ curl -L https://github.com/docker/machine/releases/download/v0.16.2/docker-machine-`uname -s`-`uname -m` >/tmp/docker-machine &&
    chmod +x /tmp/docker-machine &&
    sudo cp /tmp/docker-machine /usr/local/bin/docker-machine


# ubuntu wsl에서 실행 gcloud 인증 명령 
gcloud auth application-default login   

# ubuntu swl에서 실행 GCP에 Docker-Machine 설치 
# 서울리전 asia-northeast3
docker-machine create --driver google  --google-zone asia-northeast3-a  --google-project kmsbox --google-machine-type f1-micro --google-tags 'http-server' --google-machine-image https://www.googleapis.com/compute/v1/projects/ubuntu-os-cloud/global/images/ubuntu-2204-jammy-v20220712a gcp-host


#gcp 컴퓨트 이미지 리스트 확인 
gcloud compute images list --filter="name=ubuntu" 
> https://cloud.google.com/sdk/gcloud/reference/compute/images/list
